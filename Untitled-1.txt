SELECT
    book_id,
    book_name,
    book_descr,
    book_price,
    genre_name,
GROUP_CONCAT(DISTINCT genre.genre_name ORDER BY genre.genre_name ASC) AS genre
FROM
    book_genre
LEFT JOIN book ON book.book_id = book_genre.book_id_conn
LEFT JOIN genre ON genre.genre_id = book_genre.genre_id_conn  
GROUP BY genre.genre_name



SELECT
    book_id,
    book_name,
    book_descr,
    book_price,
    author_name
FROM
    book_author
LEFT JOIN book ON book.book_id = book_author.book_id_conn
LEFT JOIN author ON author.author_id = book_author.author_id_conn


SELECT *
FROM book_author
WHERE  book_id_conn = 1;
